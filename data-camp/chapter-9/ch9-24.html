require(mosaic) require(ggformula) require(supernova) require(tidyverse) Fingers <- supernova::Fingers custom_seed(25) Thumb.stats <- favstats(~ Thumb, data = Fingers) # change the population mean to 65 SDoM <- do(10000) * mean(rnorm(157, Thumb.stats$mean, Thumb.stats$sd)) # this creates a histogram of this SDoM gf_histogram(~ mean, data = SDoM, bins = 100, color = "blue") %>% gf_vline(xintercept = 65, color = "darkblue")%>% gf_labs(title = "SDoM, n=157, M=65") # change the population mean to 65 SDoM <- do(10000) * mean(rnorm(157, mean = 65, Thumb.stats$sd)) # this creates a histogram of this SDoM gf_histogram(~ mean, data = SDoM, bins = 100, color = "blue") %>% gf_vline(xintercept = 65, color = "darkblue")%>% gf_labs(title = "SDoM, n=157, M=65") test_function("gf_histogram", args = c("data", "bins", "color")) test_function("gf_vline") test_function("gf_labs") test_error()
