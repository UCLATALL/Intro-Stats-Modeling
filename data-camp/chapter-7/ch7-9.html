require(mosaic) require(ggformula) Fingers <- read.csv(file="https://raw.githubusercontent.com/UCLATALL/intro-stats-modeling/master/datasets/fingers.csv", header=TRUE, sep=",") Sex.model <- lm(Fingers$Thumb ~ Fingers$Sex) Fingers$Sex.resid <- resid(Sex.model) # this code generates histograms with lines to represent the mean Thumb length of each Sex group # modify the code to add lines to represent the mean Sex.resid of each group Thumb.stats <- favstats(Thumb ~ Sex, data = Fingers) gf_histogram(..density.. ~ Thumb, data = Fingers) %>% gf_facet_grid(Sex ~ .) %>% gf_vline(xintercept = ~mean, color = "blue", data = Thumb.stats) Thumb.stats <- favstats(Thumb ~ Sex, data = Fingers) gf_histogram(..density.. ~ Thumb, data = Fingers) %>% gf_facet_grid(Sex ~ .) %>% gf_vline(xintercept = ~mean, color = "blue", data = Thumb.stats)
